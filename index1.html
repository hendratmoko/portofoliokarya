const SS_ID = '1zILB3V4dl1f-7xjV_-WYoPQWvGNT8Xwqw0D52boogwk';

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(SS_ID);

  if (data.action === 'register') {
    const sheet = ss.getSheetByName('Users') || ss.insertSheet('Users');
    if (sheet.getLastRow() === 0) sheet.appendRow(['Nama','Status','Jurusan','Kontak','Kode','Tanggal']);
    const code = generateCode(data.nama);
    sheet.appendRow([data.nama, data.status, data.jurusan, data.kontak, code, new Date().toISOString()]);
    return ContentService.createTextOutput(JSON.stringify({success:true, code})).setMimeType(ContentService.MimeType.JSON);
  }

  if (data.action === 'login') {
    const sheet = ss.getSheetByName('Users');
    if (!sheet) return ContentService.createTextOutput(JSON.stringify({success:false})).setMimeType(ContentService.MimeType.JSON);
    const rows = sheet.getDataRange().getValues();
    const user = rows.find(r => r[4] === data.code);
    if (user) return ContentService.createTextOutput(JSON.stringify({success:true, nama:user[0], status:user[1], jurusan:user[2]})).setMimeType(ContentService.MimeType.JSON);
    return ContentService.createTextOutput(JSON.stringify({success:false})).setMimeType(ContentService.MimeType.JSON);
  }

  if (data.action === 'submit') {
    const sheet = ss.getSheetByName('Karya') || ss.insertSheet('Karya');
    if (sheet.getLastRow() === 0) sheet.appendRow(['Judul','Deskripsi','Kategori','Mapel','Tahun','Link','Publish','Nama','Kode','Tanggal','Rating']);
    sheet.appendRow([data.judul, data.deskripsi, data.kategori, data.mapel, data.tahun, data.link, data.publish, data.nama, data.kode, new Date().toISOString(), 0]);
    return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON);
  }

  if (data.action === 'getWorks') {
    const sheet = ss.getSheetByName('Karya');
    if (!sheet || sheet.getLastRow() < 2) return ContentService.createTextOutput(JSON.stringify({success:true, data:[]})).setMimeType(ContentService.MimeType.JSON);
    const rows = sheet.getDataRange().getValues().slice(1);
    const works = rows.filter(r => r[6] === true || r[6] === 'TRUE').map(r => ({judul:r[0],deskripsi:r[1],kategori:r[2],mapel:r[3],tahun:r[4],link:r[5],nama:r[7],tanggal:r[9],rating:r[10]}));
    return ContentService.createTextOutput(JSON.stringify({success:true, data:works})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SS_ID);
  const sheet = ss.getSheetByName('Karya');
  if (!sheet || sheet.getLastRow() < 2) return ContentService.createTextOutput(JSON.stringify({success:true, data:[]})).setMimeType(ContentService.MimeType.JSON);
  const rows = sheet.getDataRange().getValues().slice(1);
  const works = rows.filter(r => r[6] === true || r[6] === 'TRUE').map(r => ({judul:r[0],deskripsi:r[1],kategori:r[2],mapel:r[3],tahun:r[4],link:r[5],nama:r[7],tanggal:r[9],rating:r[10]}));
  return ContentService.createTextOutput(JSON.stringify({success:true, data:works})).setMimeType(ContentService.MimeType.JSON);
}

function generateCode(nama) {
  const parts = nama.trim().split(' ');
  let prefix = '';
  if (parts.length >= 2) prefix = (parts[0][0] + parts[1][0] + parts[1][1]).toUpperCase();
  else prefix = nama.substring(0, 3).toUpperCase();
  const num = Math.floor(Math.random() * 900) + 100;
  return prefix + num;
}
